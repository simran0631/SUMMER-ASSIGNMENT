#include<stdio.h>
int main()
{
    int i,j,n;
    printf("Enter the number of rows:");
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    {
        //Print spaces
        for(j=1;j<=n-1;j++)
    {
        printf("");
    }
    //Print stars
    for(j=1;j<=(2*i-1);j++)
    {
    printf("*");
    }
    printf("\n");
    }
    return 0;
}
