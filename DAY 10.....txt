#include<stdio.h>
int main()
{
    int i,j,space;
    for(i=1;i<=5;i++){
        //Print spaces
        for(space=1;space<=5-i;space++){
            printf("");
        }
        //Print increasinG characters
        for(j=0;j<i;j++){
        printf("%c",'A'+j);
    }
    //Print decreasing numbers
    for(j=i-2;j>=0;j--){
        printf("%c",'A'+j);
    }
    printf("\n");
    }
    return 0;
}
