#include<stdio.h>
int main()
{
    int arr[100],n,i,sum=0;
    float avg;
    printf("Enter the size of array:");
    scanf("%d",&n);
    printf("Enter %d elements:\n",n);
    for(i=0;i<n;i++)
    {
        scanf("%d",&arr[i]);
        sum=sum+arr[i];
    }
    avg=(float)sum/n;
    printf("Sum=%d\n",sum);
    printf("Average=%2f",avg);
    return 0;
}
