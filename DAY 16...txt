#include<stdio.h>
int main()
{
    int arr[]={2,3,4,2,5,2,3,4,4,4};
    int n=sizeof(arr)/sizeof(arr[0]);
    int maxFreq=0,maxElement;
    for(int i=0;i<n;i++){
        int count=1;
        for(int j=i+1;j<n;j++){
            if(arr[i]==arr[j]){
                count++;
            }
        }
        if(count>maxFreq){
            maxFreq=count;
            maxElement=arr[i];
        }
    }
    printf("Element with Maximum Frequency=%d\n",maxElement);
    printf("Frequency=%d\n",maxFreq);
    return 0;
}
