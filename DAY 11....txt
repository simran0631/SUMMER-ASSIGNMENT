#include<stdio.h>
//Function to check prime
int isPrime(int n)
{
    int i;
    if(n<=1)
    return 0;
    for(i=2;i<=n/2;i++)
    {
        if(n%i==0)
        return 0;
    }
    return 1;
}
int main()
{
    int num;
    printf("Enter a number:");
    scanf("%d%d",&num);
    if(isPrime(num))
    printf("%d is a prime number",num);
    else
    printf("%d is not a Prime number.",num);
    return 0;
}
