#include<stdio.h>
//Function to find maximum of two numbers
int maximum(int a,int b)
{
    if(a>b)
    return a;
    else
    return b;
}
int main()
{
    int num1,num2,result;
    printf("Enter two numbers:");
    scanf("%d%d",&num1,&num2);
    max=maximum(num1,num2);
    printf("Maximum=%d",max);
    return 0;
}
